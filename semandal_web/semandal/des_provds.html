Este método muestra todos los pueblos de una provincia con el nombre que le indiquemos. El nombre de la provincia puede estar en mayúsculas o minúsculas y : - No se acentúan las palabras

{
  • id_provincia:"integer",
  • dsprovincia:"string",
  • npueblos:"integer",
  • pueblos:[
      • {
    • id:"int",
    • dspueblo:"string",
    • coordenadas:{
      • longitud:"double",
      • longitud:"double"
      }
    • url:"string",
    • opencms:"boolean",
    • habitantes:"integer",
    • deuda:"double",
    • deudaxhab:"double",
    • fecha_inscripcion:"string",
    • superficie:"double",
    • wiki:"string",
    • cp:"integer"
    • n_noticias:"integer"
      • }
    ]
}
